def count_data(Y_pred, Y_train):
    """scikit-learnpandas"""
    true_positive, false_positive, true_negative, false_negative  = 0, 0, 0, 0
    for i,_ in enumerate(Y_pred):
        if(Y_pred[i] == Y_train[i]):
            if Y_pred[i] == 1:
                true_positive += 1
            else:
                true_negative += 1
        else:
            if Y_pred[i] == 1:
                false_positive += 1
            else:
                false_negative += 1
            
    
    return true_positive, false_positive, true_negative, false_negative

def calc_fscore(beta, tp, fp, tn, fn):
    precision = tp * 1.0/(tp + fp)
    recall = tp * 1.0/(tp + fn)
    return (1 + beta ** 2) * precision * recall /(beta ** 2 * precision + recall)

def summary_importance(importances, X_train, raw_columns):
    """
    one-hotone-hot()
    """
    col = X_train.columns
    i_c = zip(importances, col)
    sum_importances = dict()
    
    for col in raw_columns:
        sum_importances[col] = 0
    for pair in i_c:
        col_names = pair[1].split()
        col_name = col_names[0]
        if len(col_names) > 1:
            col_name = col_name[:-1]
        sum_importances[col_name] += pair[0]
        
    res = list()
    for i,col in enumerate(raw_columns):
        res.append((sum_importances[col], col, i))
    return res
